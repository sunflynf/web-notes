## React

- `useState(initialValue)` can put **callback** for setting default value

<Tabs>
    <TabItem value='old-way' label="Normal">
        ```jsx
        const Modal = ({show, onHide, children, ...rest}) => {
            const [config, setConfig] = useState({});

            useEffect(() => {
                const newConfig = buildConfig(rest);
                setConfig(newConfig);
            }, []);

            return (
                <Modal
                    show={show}
                    onHide={onHide}
                    {...config}>
                    {children}
                </Modal>
            );
        }
        ```
    </TabItem>
    <TabItem value='new-way' label='Callback'>
        ```jsx
        const Modal = ({show, onHide, children, ...rest}) => {
            const [config, setConfig] = useState(() => buildConfig(rest));
            // Yeap, it's done!

            return (
                <Modal
                    show={show}
                    onHide={onHide}
                    {...config}>
                    {children}
                </Modal>
            );
        }
        ```
    </TabItem>

</Tabs>
