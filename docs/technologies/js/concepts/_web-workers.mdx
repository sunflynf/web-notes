## Web Workers

Independent JavaScript works behind the sences

```js
let wwk;
function startWorker() {
    if (typeof(Worker) == "undefined") {
        // Web worker support!
        wwk = new Worker("for_worker.js", {}); // {} is optional
    }
    wwk.onmessage = (e) => { // something here! }
}
function stopWorker() {
    wwk.terminate();
    wwk = undefined; // reuse
}
```
